Tích hợp Google Maps

Khi shipper cập nhật vị trí, người mua có thể xem hành trình giao hàng trên bản đồ.

Khuyến nghị

Với khối lượng yêu cầu hiện tại, đây là một đồ án hoàn chỉnh, ước tính:

Khoảng 180–250 file PHP/CSS/JS.
Khoảng 15.000–20.000 dòng mã nguồn.
Đầy đủ CSDL MySQL.
Chạy trực tiếp trên XAMPP.

Mình có thể xây dựng toàn bộ dự án hoàn chỉnh, bao gồm cả mã nguồn, cơ sở dữ liệu, giao diện và chức năng, nhưng sẽ phải thực hiện theo từng phần trong nhiều lần trao đổi vì một câu trả lời không thể chứa toàn bộ hàng chục nghìn dòng mã. Mỗi phần sẽ có thể chạy được ngay để bạn ghép thành dự án hoàn chỉnh.

id INT AUTO_INCREMENT PRIMARY KEY,

phone VARCHAR(20),

avatar VARCHAR(255),

FOREIGN KEY(role_id)

REFERENCES roles(id)

);

CREATE TABLE product_images(

id INT AUTO_INCREMENT PRIMARY KEY,

product_id INT,

image VARCHAR(255),

FOREIGN KEY(product_id)

REFERENCES products(id)

ON DELETE CASCADE

);
CREATE TABLE carts(

id INT AUTO_INCREMENT PRIMARY KEY,

user_id INT,

created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

FOREIGN KEY(user_id)

REFERENCES users(id)

);

CREATE TABLE cart_items(

id INT AUTO_INCREMENT PRIMARY KEY,

cart_id INT,

product_id INT,

quantity DECIMAL(10,2),

price DECIMAL(12,0),

FOREIGN KEY(cart_id)

REFERENCES carts(id)

ON DELETE CASCADE,

FOREIGN KEY(product_id)

REFERENCES products(id)

);


CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    full_name VARCHAR(100) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    address TEXT NOT NULL,
    note TEXT,
    total DECIMAL(12,2) NOT NULL,
    status ENUM('Chờ xác nhận','Đã xác nhận','Đang giao','Hoàn thành','Đã hủy')
        DEFAULT 'Chờ xác nhận',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE order_details (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    product_id INT NOT NULL,
    quantity INT NOT NULL,
    price DECIMAL(12,2) NOT NULL
);

CREATE TABLE user_addresses(

id INT AUTO_INCREMENT PRIMARY KEY,

user_id INT NOT NULL,

full_name VARCHAR(100),

phone VARCHAR(20),

address TEXT,

address_detail VARCHAR(255),

latitude DECIMAL(10,7),

longitude DECIMAL(10,7),

is_default TINYINT(1) DEFAULT 0,

created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP

);

CREATE TABLE reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    product_id INT NOT NULL,
    user_id INT NOT NULL,
    rating TINYINT NOT NULL,
    comment TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    INDEX(order_id),
    INDEX(product_id),
    INDEX(user_id)
);
